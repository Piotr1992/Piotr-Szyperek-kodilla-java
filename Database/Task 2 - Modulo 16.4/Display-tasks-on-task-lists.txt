SELECT I.SUMMARY, L.NAME 
FROM ISSUES I JOIN ISSUESLISTS L ON L.ID = I.ISSUESLIST_ID;



SELECT I.SUMMARY, U.FIRSTNAME, U.LASTNAME 
FROM ISSUES I JOIN USERS U ON U.ID = I.USER_ID_ASSIGNEDTO;



SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASK_NUMBER 
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.LASTNAME, U.FIRSTNAME;
